async function search() { const q = document.getElementById('q').value.trim(); if (!q) return; const res = await fetch(`http://localhost:4040/api/search?q=${encodeURIComponent(q)}`); const data = await res.json(); // Build results list const resultsHTML = data.results .map(r => `
  • ${r}
    • `) .join(''); document.getElementById('out').innerHTML = `

    ${data.message}

    Reversed Query: ${data.flippedQuery}

      ${resultsHTML}
    `; }
